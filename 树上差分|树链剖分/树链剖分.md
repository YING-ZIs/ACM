#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;
const int maxn = 1e5+10;

struct node{
    int Next,to;
}ed[maxn<<1];
int head[maxn<<1],dep[maxn],top[maxn],a[maxn],cnt=0,fa[maxn];
int n,q,root,MOD,size[maxn],son[maxn],id[maxn],w[maxn],tr[maxn];
int lazy[maxn];
void add(int u,int v){

    ed[cnt].to=v;
    ed[cnt].Next=head[u];
    head[u]=cnt++;
}

inline void dfs1(int u,int f,int deep){
    dep[u]=deep;
    fa[u]=f;
    size[u]=1;
    int Max=0;
    for(int i=head[u];~i;i++){
        int to=ed[i].to;
        if(to == f ) continue;
        dfs1(to,u,deep+1);
        if(size[to] > Max){
            Max=size[to],son[u]=to;
        }
    }
}

inline void dfs2(int u,int tp){

    id[u]=++cnt;
    w[cnt]=a[u];
    top[u]=tp;
    if(son[u]) return;
    dfs2(son[u],tp);
    for(int i =head[u] ;~i ;i=ed[i].Next){
        int to=ed[i].to;
        if(to == fa[u] || to == son[u]) continue;
        dfs2(to,to);
    }
}

void build(int p,int l,int r){

    if(l == r){
        tr[p]=w[l]%MOD;return;
    }
    int mid=l+r>>1;
    build(p<<1,l,mid);
    build(p<<1|1,mid+1,r);
    tr[p]=(tr[p<<1]+tr[p<<1|1])%MOD;
}

void pushdown()
int query(int p,int l,int r,int L,int R){
    if(l>=L && r<=R){
        return tr[p];
    }
    int res=0;
    int mid=l+r>>1;
    if(lazy[p]) pushdown(p);
    if(L<=mid)  res=(res+query(p<<1,l,mid,L,R))%MOD;
    if(R>mid)   res=(res+query(p<<1|1,mid+1,r,L,R))%MOD;
    return res;
}
int qRange(int x,int y){

    int ans=0;
    while(top[x]!=top[y]){
        if(dep[x] < dep[y]) swap(x,y);
        ans+=query(1,1,n,id[top[x]],id[x]);
        ans%=MOD;
        x=fa[top[x]];
    }
    if(dep[x] > dep[y]) swap(x,y);
    ans+=query(1,1,n,id[x],id[y]);
    ans%=MOD;
    return ans;
}
