#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;
const int maxn = 2e5+10;

struct node{
    int Next,to;
}ed[maxn];
int head[maxn],dep[maxn],top[maxn],a[maxn],cnt=0,fa[maxn];
int n,q,root,MOD,size[maxn],son[maxn],id[maxn],w[maxn],tr[maxn<<2];
int lazy[maxn<<2];

inline void add(int u,int v){

    ed[cnt].to=v;
    ed[cnt].Next=head[u];
    head[u]=cnt++;
}

inline void dfs1(int u,int f,int deep){
    dep[u]=deep;
    fa[u]=f;
    size[u]=1;
    int Max=-1;
    for(int i=head[u];~i;i=ed[i].Next){
        int to=ed[i].to;
        if(to == f ) continue;
        dfs1(to,u,deep+1);
        size[u]+=size[to];
        if(size[to] > Max){
            Max=size[to],son[u]=to;
        }
    }
}

inline void dfs2(int u,int tp){

    id[u]=++cnt;
    w[cnt]=a[u];
    top[u]=tp;
    if(!son[u]) return;
    dfs2(son[u],tp);
    for(int i = head[u] ;~i ;i=ed[i].Next){
        int to=ed[i].to;
        if(to == fa[u] || to == son[u]) continue;
        dfs2(to,to);
    }
}

inline void build(int p,int l,int r){

    if(l == r){
        tr[p]=w[l]%MOD;return;
    }
    int mid=l+r>>1;
    build(p<<1,l,mid);
    build(p<<1|1,mid+1,r);
    tr[p]=(tr[p<<1]+tr[p<<1|1])%MOD;
}

inline void pushdown(int p,int l,int r){

    int mid=l+r>>1;
    tr[p<<1]=(tr[p<<1]+lazy[p]*(mid-l+1))%MOD;
    tr[p<<1|1]=(tr[p<<1|1]+lazy[p]*(r-mid))%MOD;
    lazy[p<<1|1]=(lazy[p<<1|1]+lazy[p])%MOD;
    lazy[p<<1]=(lazy[p<<1]+lazy[p])%MOD;
    lazy[p]=0;

}
inline int query(int p,int l,int r,int L,int R){
    if(l>=L && r<=R){
        return tr[p];
    }
    int res=0;
    int mid=l+r>>1;
    if(lazy[p]) pushdown(p,l,r);
    if(L<=mid)  res=(res+query(p<<1,l,mid,L,R))%MOD;
    if(R>mid)   res=(res+query(p<<1|1,mid+1,r,L,R))%MOD;
    return res;
}
inline int qRange(int x,int y){

    int ans=0;
    while(top[x]!=top[y]){
        if(dep[top[x]] < dep[top[y]]) swap(x,y);
        ans+=query(1,1,n,id[top[x]],id[x]);
        ans%=MOD;
        x=fa[top[x]];
    }
    if(dep[x] > dep[y]) swap(x,y);
    ans+=query(1,1,n,id[x],id[y]);
    ans%=MOD;
    return ans;
}

inline void update(int p,int l,int r,int L,int R,int x){

    if(l>=L && r <= R){
        tr[p]=(tr[p]+x*(r-l+1))%MOD;
        lazy[p]=(lazy[p]+x)%MOD;
        return;
    }
    if(lazy[p]) pushdown(p,l,r);
    int mid=l+r>>1;
    if(L<=mid)  update(p<<1,l,mid,L,R,x);
    if(R>mid)   update(p<<1|1,mid+1,r,L,R,x);
    tr[p]=(tr[p<<1]+tr[p<<1|1])%MOD;
}
inline int updRange(int x,int y,int z){

    z%=MOD;
    while(top[x]!=top[y]){
        if(dep[top[x]]<dep[top[y]]) swap(x,y);
        update(1,1,n,id[top[x]],id[x],z);
        x=fa[top[x]];
    }
    if(dep[x]>dep[y]) swap(x,y);
    update(1,1,n,id[x],id[y],z);
}

inline int qson(int x){

    return (query(1,1,n,id[x],id[x]+size[x]-1))%MOD;
}
inline void updson(int u,int c){

    update(1,1,n,id[u],id[u]+size[u]-1,c);
}

    }

        int u,v;scanf("%d%d",&u,&v);
        add(u,v),add(v,u);
    }
    dfs1(root,0,1);cnt=0;
    dfs2(root,root);
    build(1,1,n);
    while(q--){
        int opt;scanf("%d",&opt);
        if(opt == 1){
            int x,y,z;scanf("%d%d%d",&x,&y,&z);
            updRange(x,y,z);
        }
        else if(opt == 2){
            int x,y;scanf("%d %d",&x,&y);
            printf("%d\n",qRange(x,y));
        }
        else if(opt == 3){
            int x,y;scanf("%d %d",&x,&y);
            updson(x,y);
        }
        else{
            int x;scanf("%d",&x);
            printf("%d\n",qson(x));
        }
    }
    return 0;
}
